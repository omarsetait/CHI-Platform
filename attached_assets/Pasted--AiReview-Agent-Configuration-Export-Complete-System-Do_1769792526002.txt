## 2. Agent Configuration Schema

### Database Table: `agent_configs`

```sql
CREATE TABLE agent_configs (
  id VARCHAR PRIMARY KEY DEFAULT gen_random_uuid(),
  agent_id TEXT NOT NULL UNIQUE,
  agent_name TEXT NOT NULL,
  layer INTEGER NOT NULL,                      -- 1, 2, 3, 4, or 5
  model_provider TEXT DEFAULT 'OpenAI',
  model_name TEXT DEFAULT 'gpt-4o-mini',
  temperature DECIMAL(3,2) DEFAULT '0.2',
  max_tokens INTEGER DEFAULT 4096,
  system_prompt TEXT,                          -- Full prompt template
  prompt_template TEXT,                        -- Optional structured template
  prompt_version TEXT DEFAULT '1.0',
  weight DECIMAL(3,2) DEFAULT '1.0',           -- Aggregation weight
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

### Agent Function Code

```typescript
export async function runRegulatoryComplianceAgent(claim: ClaimContext): Promise<AgentSignal> {
  // Get config from database
  const config = await storage.getAgentConfig("regulatory_compliance");
  const temperature = config?.temperature ? parseFloat(config.temperature) : 0.1;
  const maxTokens = config?.maxTokens || 2000;
  const promptVersion = config?.promptVersion || "1.0";
  const storedSystemPrompt = config?.systemPrompt || DEFAULT_REGULATORY_COMPLIANCE_PROMPT;
  
  // Separate services from medications based on code_type
  const services = claim.lineItems?.filter(li => li.code_type !== 'SFDA') || [];
  const medications = claim.lineItems?.filter(li => li.code_type === 'SFDA') || [];
  
  // Query knowledge base for regulatory context (NPHIES/CCHI)
  const diagnosisContext = claim.diagnoses?.map(d => `${d.code}: ${d.desc}`).join(", ") || "";
  const serviceContext = services.map(li => `${li.code_type || 'SBS'}:${li.code}`).join(", ") || "";
  const medicationContext = medications.map(li => `SFDA:${li.code}`).join(", ") || "";
  
  const ragQuery = `NPHIES regulatory requirements and CCHI compliance for: ${claim.encounterType || "encounter"}, diagnoses: ${diagnosisContext}, SBS services: ${serviceContext}, SFDA medications: ${medicationContext}, network: ${claim.networkStatus || "unknown"}`;
  
  let ragContext: RAGContext | null = null;
  let ragPromptSection = "";
  try {
    ragContext = await getPhaseContext(1, ragQuery);
    if (ragContext.chunks.length > 0) {
      ragPromptSection = `\n\n## Regulatory Guidelines from Knowledge Base\n${formatRAGContextForPrompt(ragContext)}`;
    }
  } catch (error) {
    console.log("RAG context retrieval skipped:", error);
  }

  // Build user prompt with claim details and RAG context
  const userPrompt = `CLAIM DETAILS:
Claim ID: ${claim.claimId}
Member ID: ${claim.memberId}
...
${ragPromptSection}

Evaluate this claim for CHI regulatory compliance.`;

  const result = await callAgent(storedSystemPrompt, userPrompt, temperature, maxTokens);

  return {
    detector: "regulatory_compliance",
    signalId: `RC-${Date.now().toString(36)}`,
    riskFlag: result.riskFlag ?? false,
    severity: result.severity || "LOW",
    confidence: result.confidence ?? 0.8,
    recommendation: result.recommendation || "APPROVE",
    rationale: result.rationale || "Regulatory compliance check completed",
    evidence: result.evidence || [],
    missingInfo: result.missingInfo || [],
    modelInfo: { provider: "OpenAI", name: "gpt-4o-mini", temperature, prompt_version: promptVersion },
    isHardStop: result.isHardStop ?? false
  };
}
```

  const conflictingSignals = detectSignalConflicts(signals);
  const hasCriticalConflicts = conflictingSignals.some(c => c.severity === "CRITICAL");
  const hasLogicalConflicts = conflictingSignals.some(c => c.severity === "WARNING");

  }

  for (const signal of signals) {
    const weight = LAYER_WEIGHTS[signal.detector];
    const recScore = RECOMMENDATION_SCORES[signal.recommendation];
    const severityMult = signal.riskFlag ? SEVERITY_MULTIPLIERS[signal.severity] : 1.0;
    const confidenceWeight = signal.confidence;
    signalScore = recScore * severityMult * confidenceWeight * weight;

  // 4. Determine top recommendation based on score, risk, and conflicts
  if (hasCriticalConflicts) {
